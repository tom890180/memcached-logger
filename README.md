MySQL
```
CREATE TABLE `Log` (
	`ID` INT(10) UNSIGNED NOT NULL AUTO_INCREMENT,
	`Created` DATETIME NULL DEFAULT NULL,
	`Uptime` INT(10) UNSIGNED NOT NULL DEFAULT '0',
	`Used_Cache_Size` BIGINT(20) UNSIGNED NOT NULL DEFAULT '0',
	`Total_Connections` BIGINT(20) UNSIGNED NOT NULL DEFAULT '0',
	`Get_Cmd` BIGINT(20) UNSIGNED NOT NULL DEFAULT '0',
	`Set_Cmd` BIGINT(20) UNSIGNED NOT NULL DEFAULT '0',
	`Items_Evicted` BIGINT(20) UNSIGNED NOT NULL DEFAULT '0',
	`Bytes_Read` BIGINT(20) UNSIGNED NOT NULL DEFAULT '0',
	`Bytes_Written` BIGINT(20) UNSIGNED NOT NULL DEFAULT '0',
	`Current_Items` INT(10) UNSIGNED NOT NULL DEFAULT '0',
	`Hits` BIGINT(20) UNSIGNED NOT NULL DEFAULT '0',
	`Misses` BIGINT(20) UNSIGNED NOT NULL DEFAULT '0',
	PRIMARY KEY (`ID`) USING BTREE
)
COLLATE='utf8mb4_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=2
;

```
